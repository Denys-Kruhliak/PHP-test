CREATE TABLE `mvc01`.`users` ( `id` INT UNSIGNED NOT NULL AUTO_INCREMENT ,  `name` VARCHAR(128) NOT NULL ,  `email` VARCHAR(128) NOT NULL ,  `is_confirmed` TINYINT NULL DEFAULT NULL ,  `role` ENUM('admin','user') NOT NULL ,  `password_hash` VARCHAR(255) NOT NULL ,  `auth_token` VARCHAR(255) NOT NULL ,  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ,    PRIMARY KEY  (`id`),    UNIQUE  (`email`)) ENGINE = InnoDB;


CREATE TABLE `mvc01`.`posts` ( `id` INT UNSIGNED NOT NULL AUTO_INCREMENT ,  `name` VARCHAR(255) NOT NULL ,  `text` TEXT NOT NULL ,  `author_id` INT UNSIGNED NOT NULL ,  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ,    PRIMARY KEY  (`id`)) ENGINE = InnoDB;
